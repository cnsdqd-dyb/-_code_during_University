abstract class Student{
	String Name;
	double ChineseScore;
	double MathScore;
	
	abstract double GetAverageScore();
}

class PrimaryStudent extends Student{
	double GetAverageScore() {
		return (this.ChineseScore+this.MathScore)*0.5;
	}
	
}

class MiddleStudent extends Student{
	double EnglishScore;
	double GetAverageScore() {
		return (this.ChineseScore+this.MathScore+this.EnglishScore)/3.;
	}
	
}

```java
package base_dongyubo;

public class homework4 {
	public static void main(String[] args) {
		Student[] students = new Student[5];
		students[0] = new PrimaryStudent("小红",87,96);
		students[1] = new PrimaryStudent("小橙",67,89);
		students[2] = new MiddleStudent("小黄",67,89,90);
		students[3] = new MiddleStudent("小绿",87,89,86);
		
		Student bestOne = null;
		double bestScore=0;
		for(int i=0;i<4;i++) {
			if(students[i].GetAverageScore()>bestScore) {
				bestScore = students[i].GetAverageScore();
				bestOne = students[i];
			}
		}
		System.out.println("平均分最高学生: "+bestOne+": "+bestScore);
	}
}

abstract class Student{
	String Name;
	double ChineseScore;
	double MathScore;
	Student(String name,double c,double m){
		this.Name = name;
		this.ChineseScore = c;
		this.MathScore = m;
	}
	abstract double GetAverageScore();
}

class PrimaryStudent extends Student{
	PrimaryStudent(String name,double c,double m){
		super(name,c,m);
	}
	double GetAverageScore() {
		return (this.ChineseScore+this.MathScore)*0.5;
	}
	public String toString() {
		return "Name:"+this.Name+", ChineseScore:"+this.ChineseScore+", MathScore:"+this.MathScore;
	}
}

class MiddleStudent extends Student{
	
	double EnglishScore;
	
	MiddleStudent(String name,double c,double m,double e){
		super(name,c,m);
		this.EnglishScore = e;
	}
	
	double GetAverageScore() {
		return (this.ChineseScore+this.MathScore+this.EnglishScore)/3.;
	}
	public String toString() {
		return "Name:"+this.Name+", ChineseScore:"+this.ChineseScore+", MathScore:"+this.MathScore+", EnglishScore"+this.EnglishScore;
	}
}
